2 The backpropagation algorithm¶

1. Network function¶

$$y = \sigma(\sigma(\sigma(\bar{x}^TW_1)W_2)\bar{w}_3),$$
$$\text{where }\bar{x} \in \mathbb{R}^{5}, W_1 \in \mathbb{R}^{5x3}, W_2 \in \mathbb{R}^{4x2}, \bar{w}_3 \in \mathbb{R}^{3}, \text{with all statements including bias substitution.}$$

2. Sum squared error function¶

$$\text{loss} = \sum_{i}\frac{1}{2}(\hat{y}_i - \sigma(\sigma(\sigma(\bar{x}_i^TW_1)W_2)\bar{w}_3))^2,$$
$$\text{where }(\bar{x}_i,\hat{y}_i)\text{ is an element of the tuple set that makes up our training data.}$$

3. Gradient of the error function (w.r.t. weight vector of the output layer)¶

\begin{align} \nabla_{w_3}\text{loss}&=\sum_{i}(\hat{y}_i -\sigma(\sigma(\sigma(\bar{x}_i^TW_1)W_2)\bar{w}_3))\cdot\sigma'(\sigma(\sigma(\bar{x}_i^TW_1)W_2)\bar{w}_3) && \text{(sum and chain rules)}\\[2em] &=\sum_{i}(\hat{y}_i - y_i)\cdot\sigma'(a_2\bar{w}_3)\cdot a_2 && \text{(see slide 03.37)}\\[2em] \end{align}

4. Derivative of the logistic function¶

\begin{align} \sigma(x) &= \frac{1}{1+e^{-x}} \stackrel{\cdot \frac{e^x}{e^x}}{=} \frac{e^x}{1+e^x} && \text{(multiplying by one)}\\[2em] \sigma'(x) &= \frac{e^x\cdot (1+e^x)-e^x\cdot e^x}{(1+e^x)^2} && \text{(quotient rule)}\\ &= \frac{e^x}{(1+e^x)^2}\\ &= \frac{e^x+e^{2x}}{(1+e^x)^2}-\frac{e^{2x}}{(1+e^x)^2} && \text{(adding zero)}\\ &= \frac{e^x\cdot (1+e^x)}{(1+e^x)^2}-\frac{e^{2x}}{(1+e^x)^2}\\ &= \frac{e^x}{1+e^x}-\frac{e^{2x}}{(1+e^x)^2}\\ &= \frac{e^x}{1+e^x}(1-\frac{e^x}{1+e^x})\\ &= \sigma(x)(1-\sigma(x)) \end{align}

# 3.8 backpropagation
# the following function computes the gradient of our loss function
# X:        data matrix (data points are expected to be row vectors (length,height))
# w:        weight vector
# y_hat:    target values
def gradient_loss_function(w):
    # see 2.3 for the derivative of our loss function
    # and 2.4 for the derivative of our activation function
    y = ann(w)
    # derivative of the loss function depends on derivative of our activation function (see 2.4)
    return np.sum((y_hat - y) * -1 * (y * (1 - y)) * X, axis=1)

# to create random data batches during mini-batch and stochastic learning
import random

def train_network(learning_rate, steps, batch_size):
    # 3.9 gradient descent
    # see slide 03.63
    
    # definition of batches must be global
    global X
    X = total_data
    global y_hat
    y_hat = y_hat_init
    
    #weights = w_init
    # use this line instead of the one above in order to better observe convergence behavior
    weights = np.array([20,15])

    # store some values during training (add initials now)
    loss_list = [loss_function(weights)]
    w1_list = [weights[0]]
    w2_list = [weights[1]]

    # update weights
    for training_step in range(steps):
        # fill our mini-batches of size "batch_size" randomly from our total data
        batch_indexes = random.sample(range(total_data.shape[1]),batch_size)
        X = total_data[:,batch_indexes]
        y_hat = y_hat_init[batch_indexes]
        # update weights w.r.t. a fixed learning rate in opposite direction of the gradient
        weights = weights - learning_rate * gradient_loss_function(weights)

        loss = loss_function(weights)
        loss_list.append(loss)
        w1_list.append(weights[0])
        w2_list.append(weights[1])
        
    return weights, w1_list, w2_list, loss_list

def plot_results():
    # for general plotting always use total data to display true underlying error function
    global X
    X = total_data
    global y_hat
    y_hat = y_hat_init
    
    # we could use latex for axis labels etc.
    #plt.rc('text', usetex=True)
    #plt.rc('font', family='serif')

    fig, axes = plt.subplots(1, 2)
    ax0 = axes[0]
    ax1 = axes[1]

    # plot loss
    ax0.plot(np.arange(len(loss_list)), np.array(loss_list))
    ax0.set_xlabel("Training step")
    ax0.set_ylabel("Network loss")

    # w0 is the weight of our bias but we omitted it for this task
    w1 = np.linspace(-40,40,100)
    w2 = np.linspace(-40,40,100)

    W1, W2 = np.meshgrid(w1, w2)
    W_coords = np.stack((W1, W2), axis=2)
    Z = np.apply_along_axis(loss_function, -1, W_coords)

    # plot contour
    cp = ax1.contourf(W1, W2, Z, cmap="viridis")
    # plot contour lines
    ax1.contour(W1, W2, Z, colors="black", linestyles="dashed")
    ax1.set_xlabel("w1")
    ax1.set_ylabel("w2")
    # alternative code employing latex
    #ax1.set_xlabel(r'$w_1$')
    #ax1.set_ylabel(r'$w_2$')

    ax1.plot(w1_list, w2_list, color="red")
    ax1.annotate("start", xy=(w1_list[0], w2_list[0]), color="red", size=12.5)
    ax1.annotate("end", xy=(w1_list[-1], w2_list[-1]), color="red", size=12.5)
    # alternative code employing latex
    #ax1.annotate(r'$w_\text{\textbf{init}}$', xy=(w1_list[0], w2_list[0]), color="red", size=12.5)
    #ax1.annotate(r'$w_\text{\textbf{final}}$', xy=(w1_list[-1], w2_list[-1]), color="red", size=12.5)

    # add colorbar to figure
    fig.colorbar(cp, label='network loss')
    fig.tight_layout()
